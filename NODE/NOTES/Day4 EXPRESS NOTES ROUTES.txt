3ï¸âƒ£ What is Route Mounting?

Route mounting means connecting separate route files to the main server file.

Instead of writing all routes in server.js, we separate them.

Example:

routes/userRoutes.js
const express = require("express");
const router = express.Router();

router.get("/", (req, res) => {
  res.send("User Route Working");
});

module.exports = router;

server.js
const userRoutes = require("./routes/userRoutes");

app.use("/api/users", userRoutes);


Now route becomes:

http://localhost:5000/api/users


This is called route mounting.

Step 2: server.js

  console.log(`Server running on http://localhost:${PORT}`);

Step 3: routes/userRoutes.js
const express = require("express");
const router = express.Router();

const { getUser, createUser } = require("../controllers/userController");

router.get("/", getUser);
router.post("/", createUser);

module.exports = router;

Step 4: controllers/userController.js
exports.getUser = (req, res) => {
  console.log(req.method);
  console.log(req.url);
  res.json({ message: "User fetched successfully" });
};

exports.createUser = (req, res) => {
  console.log(req.body);  // checking request body
  res.json({
    message: "User created",
    data: req.body
  });
};
