11.Create a Login page that reads user data from localStorage and validates the credentials.

const Login = () => {
  const [login, setLogin] = useState({ email: "", password: "" });

  const handleLogin = () => {
    const users = JSON.parse(localStorage.getItem("users")) || [];
    const validUser = users.find(
      u => u.email === login.email && u.password === login.password
    );

    if (validUser) {
      localStorage.setItem("isLogin", true);
      alert("Login Success");
    } else {
      alert("Invalid Credentials");
    }
  };

  return (
    <>
      <input onChange={(e)=>setLogin({...login,email:e.target.value})}/>
      <input type="password" onChange={(e)=>setLogin({...login,password:e.target.value})}/>
      <button onClick={handleLogin}>Login</button>
    </>
  );
};

12.Create a Register form using useState and store the user data in localStorage.
const Register = () => {
  const [user, setUser] = useState({ name: "", email: "", password: "" });

  const handleSubmit = () => {
    const users = JSON.parse(localStorage.getItem("users")) || [];
    users.push(user);
    localStorage.setItem("users", JSON.stringify(users));
    alert("Registered Successfully");
  };

  return (
    <>
      <input placeholder="Name" onChange={(e)=>setUser({...user, name:e.target.value})}/>
      <input placeholder="Email" onChange={(e)=>setUser({...user, email:e.target.value})}/>
      <input type="password" placeholder="Password"
        onChange={(e)=>setUser({...user, password:e.target.value})}/>
      <button onClick={handleSubmit}>Register</button>
    </>
  );
};

13.Store multiple users in localStorage as an array of objects and display them using .map()
const Users = () => {
  const users = JSON.parse(localStorage.getItem("users")) || [];

  return (
    <>
      {users.map((user, index) => (
        <p key={index}>{user.name} - {user.email}</p>
      ))}
    </>
  );
};

14.Create Edit and Delete functionality for the stored users using array methods.
const deleteUser = (index) => {
  const users = JSON.parse(localStorage.getItem("users"));
  users.splice(index, 1);
  localStorage.setItem("users", JSON.stringify(users));
};

15.Protected Routerusing HOC
const ProtectedRoute = (Component) => {
  return () => {
    const isLogin = localStorage.getItem("isLogin");
    return isLogin ? <Component /> : <Navigate to="/login" />;
  };
};
